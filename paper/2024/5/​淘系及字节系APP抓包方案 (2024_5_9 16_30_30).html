字节系列APP

字节系列的基本都是走的自己的协议，            
正常的话抓包的话是抓不到包的            
            
这个时候需要做一下处理。            
写Xposed插件或者frida hook，这里用frida演示，

通用 frida hook脚本：

setImmediate(function() {            
           
Java.perform(function() {            
           
    var targetClass='org.chromium.CronetClient';            
           
    var methodName='tryCreateCronetEngine';            
           
    var gclass = Java.use(targetClass);            
           
    gclass[methodName].overload('android.content.Context','boolean','boolean','boolean','boolean','java.lang.String','java.util.concurrent.Executor','boolean').implementation = function(arg0,arg1,arg2,arg3,arg4,arg5,arg6,arg7) {            
           
    }            
           
})            
           
})

淘系抓包解决方案

淘系app底层网络通信的协议并不是普通的http协议，而是使用了自己实现的一套私有协议Spdy，正常情况下用charles是抓不到包的。

解决办法：

可以通过hook spdy来过掉该协议

可以写xp插件，也可以用frida做，这里以frida为例。

frida hook脚本

           
function main() {            
    Java.perform(function () {            
        var SwitchConfig = Java.use('mtopsdk.mtop.global.SwitchConfig');            
        SwitchConfig.nQ.overload().implementation = function () {            
            return false;            
        }            
    });            
}            
           
setImmediate(main)            
           
// frida -U 淘宝 --no-pause -l hook_spdy.js
